const path = require("path");
const fs = require("fs");
const globalConfigPath = path.join(__dirname, "globalConfig.json");
const { MongoMemoryServer } = require("mongodb-memory-server");
 
export default async function Setup(){
 
  //create memory server instance
  this.mongod = new MongoMemoryServer({
    autoStart: false
  });
 
  const mongoConfig = {
    mongoDBName: "mockdb",
    mongoUri: await mongod.getConnectionString()
  };
 
  this.startMongo(mongod);
  this.writeMongoConfig(mongoConfig);
 
  //reference to db for teardown
  global.__MONGOD__ = this.mongod;
};
 
Setup.prototype.startMongo = async mongod => {
  console.log("ummm");
  if (!mongod.isRunning) {
    await mongod.start();
    console.log("setup.mongod.isRunning,", mongod.isRunning);
  }
};
 
export const foo = () => 'foo';
 
Setup.prototype.writeMongoConfig = config => {
  fs.writeFileSync(globalConfigPath, JSON.stringify(config));
  return true;
};
